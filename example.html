AngularJS-Filters

Basic Use

{{ name }} (filter: startsWith) => {{ name | startsWith:'/' }}

{{ name }} (filter: endsWith) => {{ name | endsWith:'/' }}

{{ name }} (filter: length) => {{ name | length }}

{{ name }} (filter: wordsLength) => {{ name | wordsLength }}

{{ name }} (filter: limit:10:'!!!') => {{ name | limit:10:'!!!' }}

{{ name }} (filter: limit) => {{ name | limit }}

{{ name }} (filter: words:1:'......') => {{ name | words:1:'......' }}

{{ name }} (filter: words) => {{ name | words }}

{{ name }} (filter: title) => {{ name | title }}

{{ name }} (filter: slug) => {{ name | slug }}

{{ name }} (filter: slug:'_') => {{ name | slug:'_' }}

{{ name }} (filter: snake) => {{ name | snake }}

{{ name }} (filter: camel) => {{ name | camel }}

{{ name }} (filter: studly) => {{ name | studly }}

Advanced Use

{{ name }} (filters: title, wordsLength) => {{ name | title | words }}
